my.chisq <- function( x, y ){
  n00 <- sum( (x==0) & (y==0) )
  n11 <- sum( (x==1) & (y==1) )
  n01 <- sum( (x==0) & (y==1) )
  n10 <- sum( (x==1) & (y==0) )
  cat("Table values:", n00, " ", n01, " ", n10, " ", n11, "\n")
  nx0 <- sum( x==0)
  nx1 <- sum( x==1)
  ny0 <- sum( y==0)
  ny1 <- sum( y==1)
  cat(nx0, " ", nx1, " ", ny0, " ", ny1, "\n")
  total = sum(nx0, nx1)
  e00 <- (nx0*ny0) / total
  e01 <- (nx0*ny1) / total
  e10 <- (nx1*ny0) / total
  e11 <- (nx1*ny1) / total
  cat("Expected values:", e00, " ", e01, " ", e10, " ", e11, "\n")
  chisq <- (n00-e00)^2 / e00 + (n01-e01)^2/e01 + (n10-e10)^2/e10 + (n11-e11)^2/e11
  df <- 1
  cat("chisq: ", chisq)
  return (chisq)
}
